# Custom Colors
- Use to `colors` prop to insert your own color to be used to scale the data. Provide as many colors 
as you'd like. here we set `colors={['#fff','#feb24c','#f03b20']}`

# Code Demo

import React from 'react';
import { Heatmap } from 'react-hour-heatmap';
import { Card, CardHeader, CardContent, Divider } from "@material-ui/core";

<Heatmap
  data={[
    {date: new Date("1.1.2021 01:51"), count: 25 },
    {date: new Date("1.2.2021 14:51"), count: 4 },
    {date: new Date("1.3.2021 14:51"), count: 12 },
    {date: new Date("1.3.2021 16:51"), count: 12 },
    {date: new Date("1.3.2021 16:23"), count: 22 },
    {date: new Date("1.4.2021 16:51"), count: 102 }
  ]}
  valueColumn="count"
  dateColumn="date"
  popup={(data) => { console.log(data); return (
      <Card>
        <CardHeader title="My Custom Popup" />
        <CardContent>
          Total Sum: {data.data.value}
          <Divider />
          Total Records: {data.data.minutes.filter(r => r.data.date).length}
        </CardContent>
      </Card>
  )}}
/>

```jsx
import React from "react";
import { Heatmap } from "react-hour-heatmap";

export default function App() {
  return (
    <div className="App">
      <Heatmap
        data={[
          {date: new Date("1.1.2021 01:51"), count: 25 },
          {date: new Date("1.2.2021 14:51"), count: 4 },
          {date: new Date("1.3.2021 14:51"), count: 12 },
          {date: new Date("1.3.2021 16:51"), count: 12 },
          {date: new Date("1.3.2021 16:23"), count: 22 },
          {date: new Date("1.4.2021 16:51"), count: 102 }
        ]}
        valueColumn="count"
        dateColumn="date"
        popup={(data) => (
            <Card>
              <CardHeader title="My Custom Popup" />
              <CardContent>
                Total Sum: {data.data.value}
                <Divider />
                Total Records: {data.data.minutes.filter(r => r.data.date).length}
              </CardContent>
            </Card>
        )}
/>
    </div>
  );
}

```
